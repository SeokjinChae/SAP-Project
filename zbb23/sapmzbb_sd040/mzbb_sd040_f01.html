*&---------------------------------------------------------------------*
*& Form get_update_data
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM get_update_data .

  CALL METHOD go_grid02->get_selected_rows
    IMPORTING
      et_index_rows = gt_rows.

  DESCRIBE TABLE gt_rows LINES sy-tfill.

  IF sy-tfill = 0.
    MESSAGE i006.     " 수정할 행을 선택하세요'.
    EXIT.
  ELSEIF sy-tfill = 1.
    LOOP AT gt_rows INTO gs_row.
      READ TABLE gt_sd030 INTO gs_sd030 INDEX gs_row-index.
    ENDLOOP.

    SELECT SINGLE *
      FROM zbbt_sd030
      INTO CORRESPONDING FIELDS OF zbbt_sd030
      WHERE subno EQ gs_sd030-subno
      AND   itemno EQ gs_sd030-itemno.

*     추가 후 저장 누르지 않고 해당 행을 수정할 경우
    IF zbbt_sd030-matno IS INITIAL.
      MESSAGE e000 WITH '행 추가가 반영되지 않았습니다. 수정 전 저장을 눌러주세요.'.
    ENDIF.

*     삭제 후 저장 누르지 않고 해당 행을 수정할 경우
    IF gs_sd030-matno IS NOT INITIAL AND gs_sd030-light EQ '1'.
      MESSAGE e000 WITH '행 삭제가 반영되지 않았습니다. 수정 전 저장을 눌러주세요.'.
    ENDIF.

    CALL SCREEN 120 STARTING AT 10  10.   "수정 팝업
*
  ELSE.
    MESSAGE i026.     "수정할 하나의 행을 선택하세요.
  ENDIF.

ENDFORM.

FORM delete_data .
  DATA : lv_answer.
  DATA: lt_del TYPE TABLE OF zbbt_sd030,
        ls_del TYPE zbbt_sd030.
  DATA : lv_tabix TYPE sy-tabix.

  CLEAR : lt_del, gt_rows.

  CALL METHOD go_grid02->get_selected_rows
    IMPORTING

  DESCRIBE TABLE gt_rows LINES sy-tfill.

  SORT gt_rows BY index DESCENDING.

  IF sy-tfill = 0.
    MESSAGE i003. " '항목을 선택하세요'.
    EXIT.
  ELSE.
    CALL FUNCTION 'POPUP_TO_CONFIRM'

        start_row             = 15
      IMPORTING
        answer                = lv_answer
      EXCEPTIONS
        OTHERS                = 1.

    IF lv_answer EQ 1 .    "네 선택
      LOOP AT gt_rows INTO gs_row.
        CLEAR gs_sd030.
        lv_tabix = gs_row-index.

        LOOP AT gt_sd030 INTO gs_sd030 FROM lv_tabix TO lv_tabix.

          " 수정 후 저장 누르지 않고 해당 행을 삭제할 경우

          ENDIF.

          gs_sd030-light = '1'.
          gs_sd030-color = 'C' && col_negative && '10'.
          MODIFY gt_sd030 FROM gs_sd030 TRANSPORTING light color.
          APPEND gs_sd030 TO gt_del.
        ENDLOOP.

      ENDLOOP.

      MESSAGE s000 WITH '데이터 삭제에 성공했습니다. 저장을 위해 Save 눌러주세요'.

    ELSEIF lv_answer EQ 2. "아니오 선택

      MESSAGE s025. "데이터 삭제를 취소했습니다.

    ENDIF.

  ENDIF.

FORM create_data .

  CLEAR gs_line.


  gs_line-subno = zbbt_sd030-subno.
  gs_line-itemno = gs_sd030-itemno + 10.
  gs_line-matno = zbbt_sd030-matno.

  gs_line-unit = zbbt_sd030-unit.

  MOVE-CORRESPONDING gs_line TO gs_sd030.

  SELECT SINGLE * FROM zbbt_mm020
    INTO gs_mm020
    WHERE matno = gs_sd030-matno.

  gs_sd030-matnm = gs_mm020-matnm.

  gs_sd030-light = '2'.   "Red
  gs_sd030-color = 'C' && col_group && '10'.
  APPEND gs_sd030 TO gt_sd030.


  IF sy-subrc <> 0.
    MESSAGE e002.  "데이터 생성에 실패하셨습니다.
    EXIT.
  ELSE.
    MESSAGE s000 WITH '데이터 생성에 성공했습니다. 저장을 위해 Save 눌러주세요'.
  ENDIF.

  CLEAR zbbt_sd030.



ENDFORM.
