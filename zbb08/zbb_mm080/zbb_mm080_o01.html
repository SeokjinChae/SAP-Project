MODULE CREATE_AND_TRANSFER OUTPUT.

  IF GO_CONT IS INITIAL.
    PERFORM CREATE_OBJECT.
    PERFORM SET_LAYOUT.
    PERFORM SET_SORTING.
    PERFORM SET_HANDLER.

    CASE GC_MARK.
      WHEN PA_WAIT.
        PERFORM SET_FIELD_CAT.
      WHEN PA_APP.
        PERFORM SET_FIELD_CAT2.
      WHEN PA_NAPP.
        PERFORM SET_FIELD_CAT3.
    ENDCASE.

    PERFORM DISPLAY_TABLE.
  ENDIF.

ENDMODULE.
