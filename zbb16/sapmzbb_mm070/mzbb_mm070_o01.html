MODULE set_cursor OUTPUT.

  IF cursorfield = 'ZBBT_MM010-STORNM'.        "현재 커서 위치
    SET CURSOR FIELD 'ZBBT_MM010-STORTY'.   "커서 위치 이동 시킴
  ENDIF.

  IF cursorfield = 'ZBBT_MM010-STORNM'.
    SET CURSOR FIELD 'ZBBT_MM010-STORTY'.
  ENDIF.

  IF cursorfield = 'ZBBT_MM010-STORTY'.
    SET CURSOR FIELD 'ZBBT_MM010-STORADD'.
  ENDIF.

  IF cursorfield = 'ZBBT_MM010-STORADD'.
    SET CURSOR FIELD 'STEL1'.
  ENDIF.

  IF cursorfield = 'STEL1'.
    SET CURSOR FIELD 'STEL2'.
  ENDIF.

  IF cursorfield = 'STEL2'.
    SET CURSOR FIELD 'STEL3'.
  ENDIF.

  IF cursorfield = 'STEL3'.
    SET CURSOR FIELD 'ZBBT_MM010-EMPNO'.
  ENDIF.
ENDMODULE.
