FORM user_command_0100 .

    WHEN 'SAVE'.            "저장
      PERFORM save_data.
      PERFORM get_data.
      PERFORM make_data.
      PERFORM refresh_grid.
      CLEAR gv_mark.

    WHEN 'BACK'.           "뒤로가기

    WHEN 'FC1'.             "조회
      PERFORM get_data.
      PERFORM make_data.

    WHEN 'REFRESH'.         "초기화
      CLEAR zbbt_mm030-venno.
      CLEAR gt_mm030.

    WHEN 'DOWNLOAD'.       "엑셀다운로드
      PERFORM down_excel.

    WHEN 'UPLOAD'.          "엑셀업로드
      IF gv_mark = 'X'.
        MESSAGE i077 DISPLAY LIKE 'E'.
      ELSE.
        PERFORM get_filenm CHANGING pa_file.
        PERFORM upload_excel.
        CLEAR pa_file.
      ENDIF.

  ENDCASE.


ENDFORM.

FORM get_data .

      FROM zbbt_mm030

FORM create_data .
  CLEAR : gs_line.
  DATA : lv_answer.
  DATA : lv_text      TYPE dd07t-ddtext,
         lv_value(10) TYPE c.

  IF zbbt_mm030-vennm IS NOT INITIAL AND zbbt_mm030-banknm IS NOT INITIAL
    AND zbbt_mm030-deponm IS NOT INITIAL AND zbbt_mm030-baccno IS NOT INITIAL
    AND vreg1 IS NOT INITIAL AND  vreg2 IS NOT INITIAL AND  vreg3 IS NOT INITIAL
    AND zbbt_mm030-vpicnm IS NOT INITIAL AND  zbbt_mm030-venadd IS NOT INITIAL
    AND vtel1 IS NOT INITIAL AND  vtel2 IS NOT INITIAL AND  vtel3 IS NOT INITIAL.

    "생성 전 확인
    CALL FUNCTION 'POPUP_TO_CONFIRM'
      EXPORTING
        titlebar              = '거래처 마스터 생성'
        text_question         = '거래처 데이터를 생성하시겠습니까?'
        text_button_1         = '네'
        icon_button_1         = 'ICON_SYSTEM_OKAY'
        text_button_2         = '아니오'
        icon_button_2         = 'ICON_SYSTEM_CANCEL'
        default_button        = '1'
        display_cancel_button = ' '
        start_column          = 25
        start_row             = 15
      IMPORTING
        answer                = lv_answer
      EXCEPTIONS
        OTHERS                = 1.

    IF lv_answer EQ 1 .    "네 선택
      gs_line-vennm = zbbt_mm030-vennm.
      gs_line-banknm = zbbt_mm030-banknm.
      gs_line-deponm = zbbt_mm030-deponm.
      gs_line-baccno = zbbt_mm030-baccno.

      "사업자등록번호 합치기
      CONCATENATE vreg1 '-' vreg2 '-' vreg3 INTO gv_vregno.
      gs_line-vregno = gv_vregno.

      gs_line-vpicnm = zbbt_mm030-vpicnm.
      gs_line-venadd = zbbt_mm030-venadd.

      "전화번호 합치기
      CONCATENATE vtel1 '-' vtel2 '-' vtel3 INTO gv_vtel.
      gs_line-ventel = gv_vtel.

      MOVE-CORRESPONDING gs_line TO gs_mm030.

      "은행명 txt 추가
      lv_value = gs_mm030-banknm.
      CALL FUNCTION 'STF4_GET_DOMAIN_VALUE_TEXT'
        EXPORTING
          iv_domname      = 'ZBBD_BANKNM'
          iv_value        = lv_value
        IMPORTING
          ev_value_text   = lv_text
        EXCEPTIONS
          value_not_found = 1
          OTHERS          = 2.

      gs_mm030-banknmtxt = lv_text.

      "행 추가시 기본 Light/행 색상 지정
      gs_mm030-light = '1'.   "Red
      gs_mm030-color = 'C' && col_group && '10'.


      IF sy-subrc <> 0.
        MESSAGE e002.     "데이터 생성에 실패하셨습니다.
        EXIT.
      ELSE.
        APPEND gs_mm030 TO gt_mm030.
        MESSAGE s095 WITH '추가'.   "데이터베이스에 최종 &위해 SAVE가 필요합니다.
      ENDIF.

      CLEAR zbbt_mm030.

    ELSEIF lv_answer EQ 2. "아니오 선택
      MESSAGE s097 WITH '거래처'.
    ENDIF.

  ELSE.
    MESSAGE s096 DISPLAY LIKE 'E'.   "모든 필드에 정보 입력이 필요합니다.
    LEAVE TO SCREEN 110.
  ENDIF.



ENDFORM.

FORM update_data .
  DATA : lv_venno TYPE zbbt_mm030-venno.
  CLEAR gs_line.

  "선택한 행에 대해 거래처 번호 가져오기
  lv_venno =  zbbt_mm030-venno.

  LOOP AT gt_mm030 INTO gs_mm030 WHERE venno = lv_venno.
    gs_mm030-vennm = zbbt_mm030-vennm.
    gs_mm030-banknm = zbbt_mm030-banknm.
    gs_mm030-deponm = zbbt_mm030-deponm.
    gs_mm030-baccno = zbbt_mm030-baccno.
    gs_mm030-vregno = zbbt_mm030-vregno.
    gs_mm030-vpicnm = zbbt_mm030-vpicnm.
    gs_mm030-venadd = zbbt_mm030-venadd.
    CONCATENATE vtel1 '-' vtel2 '-' vtel3 INTO gv_vtel.
    gs_mm030-ventel = gv_vtel.

    "수정 light, 행 색상 변경
    gs_mm030-light = '2'.   "yellow
    gs_mm030-color = 'C' && col_total && '00'.
    MODIFY gt_mm030 FROM gs_mm030
    TRANSPORTING venno banknm deponm baccno vregno vpicnm venadd ventel light color.

  ENDLOOP.

  IF sy-subrc = 0.
    MESSAGE s095 WITH '수정'.    "데이터베이스에 최종 수정위해 SAVE가 필요합니다.
  ELSE.
    MESSAGE e031.     "데이터 수정을 실패했습니다.
  ENDIF.

ENDFORM.

FORM delete_data .
  DATA : lv_answer.
  DATA : lt_del TYPE TABLE OF zbbt_mm030,
         ls_del TYPE zbbt_mm030.
  DATA : lv_tabix TYPE sy-tabix.

  CLEAR : lt_del, gt_rows.

  CALL METHOD go_grid->get_selected_rows
    IMPORTING
      et_index_rows = gt_rows.
  DESCRIBE TABLE gt_rows LINES sy-tfill.

  SORT gt_rows BY index DESCENDING.


  IF sy-tfill = 0.
    MESSAGE i003.       " 항목을 선택하세요.
    EXIT.
  ELSE.
    CALL FUNCTION 'POPUP_TO_CONFIRM'
      EXPORTING
        titlebar              = '삭제'
        text_question         = '데이터를 삭제 하시겠습니까?'
        text_button_1         = '네'
        icon_button_1         = 'ICON_SYSTEM_OKAY'
        text_button_2         = '아니오'
        icon_button_2         = 'ICON_SYSTEM_CANCEL'
        default_button        = '1'
        display_cancel_button = ' '
        start_column          = 25
        start_row             = 15
      IMPORTING
        answer                = lv_answer
      EXCEPTIONS
        OTHERS                = 1.

    IF lv_answer EQ 1 .    "네 선택
      CLEAR gs_mm030.
      LOOP AT gt_rows INTO gs_row.
        lv_tabix = gs_row-index.
        LOOP AT gt_mm030 INTO gs_mm030 FROM lv_tabix TO lv_tabix.

*    "<예외처리>
*    " 1) 수정 후 저장 누르지 않고 해당 행을 삭제할 경우
          IF gs_mm030-light = '2'.
            MOVE-CORRESPONDING gs_mm030 TO ls_del.
            MODIFY zbbt_mm030 FROM ls_del.
*            MESSAGE e069 WITH '수정' '삭제' .
          ENDIF.
*    " 2) 추가 후 저장 누르지 않고 해당 행을 삭제할 경우
          IF gs_mm030-venno IS INITIAL AND gs_mm030-light = '1'.
            MESSAGE i098. "'데이터베이스에 반영되지 않은 데이터입니다. 삭제 전 저장을 눌러주세요.'
            LEAVE SCREEN.
          ENDIF.

          "삭제 light, 행 색상 변경
          gs_mm030-light = '1'.
          gs_mm030-color = 'C' && col_negative && '10'.
          MODIFY gt_mm030 FROM gs_mm030 TRANSPORTING light color.
          APPEND gs_mm030 TO gt_del.
        ENDLOOP.

      ENDLOOP.
      MESSAGE s095 WITH '삭제'.  "'데이터베이스에 최종 삭제위해 SAVE가 필요합니다.'.

    ELSEIF lv_answer EQ 2. "아니오 선택

      MESSAGE s025. "데이터 삭제를 취소했습니다.

    ENDIF.

  ENDIF.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form save_data
*&---------------------------------------------------------------------*
*& 행 추가, 수정, 삭제를 DB 테이블에 최종 반영 한다. (최종 저장)
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM save_data .
  DATA : lv_answer.
  DATA: lt_del TYPE TABLE OF zbbt_mm030,
        ls_del TYPE zbbt_mm030.

  CLEAR : gs_mm030.

*  "변경 확인 및 DATA_CHANGED 이벤트 발생
*  DATA : lv_valid,
*         lv_refresh VALUE 'X'.
*
*  CALL METHOD go_grid->check_changed_data
*    IMPORTING
*      e_valid   = lv_valid
*    CHANGING
*      c_refresh = lv_refresh.

  CALL FUNCTION 'POPUP_TO_CONFIRM'
    EXPORTING
      titlebar              = '저장확인'
      text_question         = '변경사항을 최종 저장 하시겠습니까?'
      text_button_1         = '네'
      icon_button_1         = 'ICON_SYSTEM_OKAY'
      text_button_2         = '아니오'
      icon_button_2         = 'ICON_SYSTEM_CANCEL'
      default_button        = '1'
      display_cancel_button = ' '
      start_column          = 25
      start_row             = 15
    IMPORTING
      answer                = lv_answer
    EXCEPTIONS
      OTHERS                = 1.


  CASE lv_answer.
    WHEN 1.              "네 선택
      "1) 행 삭제
      IF gt_del IS NOT INITIAL.
        MOVE-CORRESPONDING gt_del TO lt_del.

        IF lt_del IS NOT INITIAL.
          LOOP AT lt_del INTO ls_del.
            ls_del-delflag = 'X'.
            " 수정 Timestamp (삭제)
            ls_del-update_dt = sy-datum.
            ls_del-update_tm = sy-uzeit.
            ls_del-uuser = sy-uname.
            MODIFY lt_del FROM ls_del.
          ENDLOOP.
          MODIFY zbbt_mm030 FROM TABLE lt_del.
          REFRESH gt_del.
        ENDIF.
      ELSE.

        "2) 행 추가, 수정
        LOOP AT gt_mm030 INTO gs_mm030.
          IF gs_mm030-light = '2'.                  "행 수정
            " 수정 Timestamp
            gs_mm030-update_dt = sy-datum.
            gs_mm030-update_tm = sy-uzeit.
            gs_mm030-uuser = sy-uname.

          ELSEIF gs_mm030-light = '1'.              "행 추가
            " 넘버레인지
            PERFORM get_info_numberrange.

            " 추가 Timestamp
            gs_mm030-create_dt = sy-datum.
            gs_mm030-create_tm = sy-uzeit.
            gs_mm030-cuser = sy-uname.
