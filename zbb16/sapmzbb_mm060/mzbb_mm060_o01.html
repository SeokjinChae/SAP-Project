*&---------------------------------------------------------------------*
*& Module SET_CURSOR OUTPUT
*&---------------------------------------------------------------------*
*& 현재 커서 위치를 받아 커서 위치를 다음으로 이동 시킴
*&---------------------------------------------------------------------*
MODULE set_cursor OUTPUT.

  IF cursorfield = 'ZBBT_MM030-VENNM'.        "현재 커서 위치
    SET CURSOR FIELD 'ZBBT_MM030-BANKNM'.   "커서 위치 이동 시킴
  ENDIF.


  IF cursorfield = 'ZBBT_MM030-BANKNM'.
    SET CURSOR FIELD 'ZBBT_MM030-DEPONM'.
  ENDIF.


  IF cursorfield = 'ZBBT_MM030-DEPONM'.
    SET CURSOR FIELD 'ZBBT_MM030-BACCNO'.
  ENDIF.


  IF cursorfield = 'ZBBT_MM030-BACCNO'.
    SET CURSOR FIELD 'VREG1'.
  ENDIF.


  IF cursorfield = 'VREG1'.
    SET CURSOR FIELD 'VREG2'.
  ENDIF.


  IF cursorfield = 'VREG2'.
    SET CURSOR FIELD 'VREG3'.
  ENDIF.


  IF cursorfield = 'VREG3'.
    SET CURSOR FIELD 'ZBBT_MM030-VPICNM'.
  ENDIF.


  IF cursorfield = 'ZBBT_MM030-VPICNM'.
    SET CURSOR FIELD 'ZBBT_MM030-VENADD'.
  ENDIF.


  IF cursorfield = 'ZBBT_MM030-VENADD'.
    SET CURSOR FIELD 'VTEL1'.
  ENDIF.


  IF cursorfield = 'VTEL1'.
    SET CURSOR FIELD 'VTEL2'.
  ENDIF.


  IF cursorfield = 'VTEL2'.
    SET CURSOR FIELD 'VTEL3'.
  ENDIF.
ENDMODULE.
