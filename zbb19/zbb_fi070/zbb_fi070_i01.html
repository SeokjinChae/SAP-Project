  save_ok = ok_code.
  CLEAR ok_code.

  CASE save_ok.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT' OR 'CANCEL'.
      PERFORM exit_program.
    WHEN 'REFRESH'.
      PERFORM refresh.
    WHEN OTHERS.
